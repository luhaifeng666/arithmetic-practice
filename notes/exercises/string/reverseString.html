# 反转字符串

# 题目

编写一个函数，其作用是将输入的字符串反转过来。输入字符串以字符数组 s 的形式给出。
不要给另外的数组分配额外的空间，你必须原地修改输入数组、使用 O(1) 的额外空间解决这一问题。

# 示例1：

输入：s = ["h","e","l","l","o"]
输出：["o","l","l","e","h"]
1
2

# 示例2：

输入：s = ["H","a","n","n","a","h"]
输出：["h","a","n","n","a","H"]
1
2

提示：

  • 1 <= s.length <= 105
  • s[i] 都是 ASCII 码表中的可打印字符

# 题解

解题思路

利用 双指针 的思想，在索引为0的位置放置一个指针，在数组末尾的位置放置一个指针，两个指针同时移动。
左指针向右移动，右指针向左移动，每次移动一位，并同时交换值。
当左指针的索引值大于等于右指针的索引值时，停止遍历

/**
 * @param s
 */
export default function(s: string[]): void {
  for (let i = 0; i < s.length; i ++) {
    if (i < s.length - 1 - i) {
      // [s[i], s[s.length - 1 - i]] = [s[s.length - 1 - i], s[i]]
      const str = s[i]
      s[i] = s[s.length - 1 - i]
      s[s.length - 1 - i] = str
    }
  }
};
